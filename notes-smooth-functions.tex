    \begin{definition}[Smooth function to a manifold]
        Let $M,$ $N$ be smooth manifolds and $f : M \to N$ a continuous function. We say $f$ is \emph{smooth} if for any chart $(U, \varphi)$ of $M$ and any chart $(V, \psi)$ of $N$, the composition 
        \[      \psi \circ f \circ \varphi ^{-1} : \varphi ( f^{-1} (V) ) \to \psi (V)                  \]
        is smooth. 
    \end{definition}

    \begin{proposition}\label{pro:manifold-chain-rule}
        Let $M_1$, $M_2$, $M_3$ be smooth manifolds and $f:M_1 \to M_2 $, $g : M_2 \to M_3$ smooth functions. Then $g \circ f : M_1 \to M_3$ is smooth.

    \end{proposition}

     

    \begin{definition}[Diffeomorphism]
        Let $M$, $N$ be smooth manifolds. A smooth homeomorphism $f : M \to N$ is called a \emph{diffeomorphism} if its inverse $f^{-1} : N \to M$ is smooth. If there is a diffeomorphism $f : M \to N$, we say that $M$ is \emph{diffeomorphic} to $N$.   
    \end{definition}

    \begin{definition}[Support]
        Let $M$ be a topological space and $f : M \to \mathbb{R}$ a continuous function. The \emph{support} of $f$, denoted by supp$(f)$ is defined as the topological closure of the set 
        \[      \{  x \in M  \, \vert \, f(x) \neq 0  \} .   \]
    \end{definition}
    
    \begin{proposition}[Bump functions]
        Let $M$ be a smooth manifold, $U \subset M$ an open set, and $p\in U$. Then there is $f \in C^{\infty}(M)$  with supp$(f) \subset U$ and $f(p) \equiv 1$ in a neighborhood of $p$. 

    
    \begin{solution}
         We start by constructing functions from $\mathbb{R}$ to $\mathbb{R}$. 
    \begin{align*}
              f_0(x)    & : = \begin{cases}
            e^{1/x}  & \text{ if } x > 0 \\
            0 & \text{ if } x \leq 0
        \end{cases}           \\
     f_1 (x) &: = f_0 (x) f_0 (1-x)  \\
      f _2 (x) &: = \int_ 0 ^x f_1 (t) \, dt      \\
      f_3 (x) & : =   [   f_2 (1)  - f_2 (x - 1 )  ] / f_2 (1)    
    \end{align*}
    This allows us to define $f_4 \in C^{\infty}(\mathbb{R}^n)$ as
    \[    f_4 (x) : = f_3 (\, \vert x \vert  \, ) .                        \]
    Now take a chart $(V, \psi ) $ around $x$ with $\psi (p) = 0$ and $\psi (U \cap V )$  containing $B_3 (0)$. With this,  define $f : M \to \mathbb{R}$ as
    \[ f(x ) : = \begin{cases}
        f_4 (\psi (x) ) & \text{ if } x \in V\\
       0 & \text{ otherwise} 
    \end{cases} \] 
    You may feel like we're done, but you still need to check that $f$ is smooth outside of $V$. This uses the fact that $M$ is Hausdorff and is left as an exercise (we finally used that hypothesis!). 
    \end{solution}
\end{proposition}

    \begin{corollary}[Extension of function]
        Let $M$ be a smooth manifold, $U \subset M$ an open set, $p \in U$, and $f \in \mathbb{C}^{\infty}(U)$. Then there is $g \in \mathbb{C}^{\infty}(M)$ with $g \equiv f$ in a neighborhood of $p$. 
    \end{corollary}


    \begin{definition}[Locally finite]
        Let $M$ be a topological space and $ \mathcal{X} = \{ X_i \} _{i \in I}$ a collection of subsets of $M$. We say $\mathcal{X}$ is \emph{locally finite} if each $p \in M$ admits an open neighborhood $U \subset M$ such that $U \cap X_i \neq \emptyset $ only for finitely many $i$'s. 
    \end{definition}

    \begin{theorem}[Partitions of unity]\label{thm:pou}
        Let $M$ be a smooth manifold and $\{ U_i \} _{i \in I} $ an open cover of $M$. Then there is a collection $\{ \rho_i \} _{i \in I}$ of functions $\rho _i \in C^{\infty }(M)$ such that
        \begin{itemize}
            \item For each $i \in I$, one has supp$(\rho_i) \subset U_i$.
            \item $\rho_i (M)\subset [0,1]$
            \item The collection $\{ $supp$(\rho_i)\}_{i \in I}$ is locally finite.
            \item For each $x \in M$, one has
            \[      \sum_{i \in I} \rho _i (x) = 1  .            \]
        \end{itemize}
    Note that the sum in the last item makes sense because only finitely many summands are non-zero. 


    \begin{solution}
        Theorem 2.23 from Lee. This uses the second-countability.
    \end{solution}
    \end{theorem}
