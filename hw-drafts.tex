\begin{exercise}
    Let $M$ be an $n$-dimensional topological manifold. Show that being compatible is an equivalence relation in the set of charts of $M$.
    \begin{hint}
        Chain rule.
    \end{hint}
\end{exercise}


\begin{exercise}
    Let $M$ be an $n$-dimensional topological manifold and $\mathcal{A}$ a smooth atlas. Show that there is a unique smooth structure $\mathcal{S}$ containing $\mathcal{A}$. Moreover, show that $\mathcal{S}$ consists precisely of the charts compatible with all charts in $\mathcal{A}$.

\begin{hint}
    Use Zorn's Lemma to show there is at least one smooth structure $\mathcal{S}$ containing $\mathcal{A}$. Let $\mathcal{S}'$ denote the set of charts compatible with all charts in $\mathcal{A}$. Show that $\mathcal{S} \subset \mathcal{S}'$ and deduce $\mathcal{S} = \mathcal{S}'$.
\end{hint}
\end{exercise}

Show that $\mathbb{R}^n$ is an $n$-dimensional smooth manifold.
